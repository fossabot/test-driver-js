svg line with array data

var arrayData = [
  [10, 130],
  [100, 60],
  [190, 160],
  [280, 10],
];
var lineGenerator = rj3.svg.line();
var path1 = lineGenerator(arrayData);
document.getElementById('path1').setAttribute('d', path1);

svg line with array object data

 var objectData = [
  { x: 10, y: 130 },
  { x: 100, y: 60 },
  { x: 190, y: 160 },
  { x: 280, y: 10 },
];
var lineGeneratorWithObject = rj3.svg.line()
  .x(function(d) { return d.x })
  .y(function(d) { return d.y });
var path2 = lineGeneratorWithObject(objectData);
document.getElementById('path2').setAttribute('d', path2);
